00119          * Suchfunktion. Erhält Informationen über das gesuchte Token, sucht in wadoku.xml und liefert eine DataTable mit Ergebnissen zurück.

00126         public DataTable search(string token, int satzNr, int tokenNr, bool absolute)
00127         {
00128             //Wenn im dataSet schoneinmal dasselbe Token mit/ohne genauer Übereinstimmung gesucht wurde, ...
00129             if (dataSet.Tables.Contains("results_" + satzNr + tokenNr + ((absolute) ? "_absolute" : "")))
00130             {
00131                 //gib die entsprechende Tabelle zurück.
00132                 return dataSet.Tables["results_" + satzNr + tokenNr + ((absolute) ? "_absolute" : "")];
00133             }
00134             //Falls noch nicht nach dem Token gesucht wurde, führe eine Suche in Wadoku aus.
00135             else
00136             {
00137                 //Erzeuge einen Knoten, der alle Suchergebnisknoten enthält,
00138                 XElement results = new XElement("results",
00139                                 //Durchsuche alle Knoten, die der Wurzelknoten in wadoku.xml enthält, ...
00140                                 (from entry in wadoku.Root.Elements().OfType<XElement>()
00141                                  //lasse validateEntry den <form>-Knoten auf die durch absolute bestimmte Weise nach dem String token durchsuchen 
00142                                  where validateEntry(entry.Element(xns + "form"), token, absolute)
00143                                  //und wähle für einen neu erstellten <result>-Knoten die folgenden Elemente aus dem Wadoku-Eintrag aus:
00144                                  select new XElement("result", new XElement("orths", entry.Element(xns + "form").Elements(xns + "orth")),   //neues xml-Element, das aus dem <form>-Knoten alle japanischen Schreibweisen aufnimmt
00145                                                                entry.Element(xns + "gramGrp"),                                              //xml-Element mit grammatischen (u.a. POS-Infos) Informationen 
00146                                                                new XElement("prons", entry.Element(xns + "form").Elements(xns + "pron")),   //neues xml-Element, das aus dem <form>-Knoten alle japanischen Umschriften aufnimmt
00147                                                                new XElement("senses", entry.Elements(xns + "sense")))));                    //neues xml-Element, das aus dem Eintrag alle <sense>-Knoten mit deutschen Bedeutungen aufnimmt      
00148                 //Wurden <result>-Knoten für gefundene Einträge erstellt, ...
00149                 DataTable outputTable = (results.HasElements) ?
00150                                             //So erstelle eine DataTable mit entsprechend formatierten Ergenissen. Erstelle ansonsten eine leere DataTable.
00151                                             (createResultTable(results, satzNr, tokenNr, absolute)) : (new DataTable());
00152                 //Gib die der Variable outputTable zugewiesene Tabelle zurück.
00153                 return outputTable;
00154             }

00167             bool entryIsValid = false;

00169
